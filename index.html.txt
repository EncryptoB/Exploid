<body>
    <h1 style="color: white; text-shadow: 2px 2px 4px rgba(0, 0, 0, 0.5);">¡Explota las Burbujas!</h1>
    <div id="score">Puntuación: 0</div>
    <div id="game-container"></div>
    <button onclick="resetGame()">Reiniciar Juego</button>

    <script>
        let score = 0;
        let isGameActive = true;
        const container = document.getElementById('game-container');
        const scoreDisplay = document.getElementById('score');

        function createBubble() {
            if (!isGameActive) return;
            
            const bubble = document.createElement('div');
            bubble.className = 'bubble';
            
            // Tamaño aleatorio entre 30 y 60 píxeles
            const size = Math.random() * 30 + 30;
            bubble.style.width = size + 'px';
            bubble.style.height = size + 'px';
            
            // Posición aleatoria
            const maxX = container.clientWidth - size;
            const maxY = container.clientHeight - size;
            bubble.style.left = Math.random() * maxX + 'px';
            bubble.style.top = container.clientHeight + 'px';
            
            container.appendChild(bubble);
            
            // Animación de subida
            let posY = container.clientHeight;
            const speed = Math.random() * 2 + 1;
            
            function moveBubble() {
                if (!isGameActive) {
                    bubble.remove();
                    return;
                }
                
                posY -= speed;
                bubble.style.top = posY + 'px';
                
                if (posY < -size) {
                    bubble.remove();
                } else {
                    requestAnimationFrame(moveBubble);
                }
            }
            
            bubble.onclick = () => {
                score += 10;
                scoreDisplay.textContent = `Puntuación: ${score}`;
                bubble.remove();
                
                // Efecto de partículas al explotar
                for (let i = 0; i < 5; i++) {
                    createParticle(
                        parseFloat(bubble.style.left) + size/2,
                        parseFloat(bubble.style.top) + size/2
                    );
                }
            };
            
            moveBubble();
        }

        function createParticle(x, y) {
            const particle = document.createElement('div');
            particle.className = 'bubble';
            particle.style.width = '10px';
            particle.style.height = '10px';
            particle.style.left = x + 'px';
            particle.style.top = y + 'px';
            particle.style.opacity = '0.7';
            
            container.appendChild(particle);
            
            const angle = Math.random() * Math.PI * 2;
            const velocity = Math.random() * 5 + 2;
            const vx = Math.cos(angle) * velocity;
            const vy = Math.sin(angle) * velocity;
            let posX = x;
            let posY = y;
            
            function moveParticle() {
                posX += vx;
                posY += vy;
                particle.style.left = posX + 'px';
                particle.style.top = posY + 'px';
                particle.style.opacity = parseFloat(particle.style.opacity) - 0.02;
                
                if (parseFloat(particle.style.opacity) > 0) {
                    requestAnimationFrame(moveParticle);
                } else {
                    particle.remove();
                }
            }
            
            moveParticle();
        }

        function resetGame() {
            score = 0;
            scoreDisplay.textContent = `Puntuación: ${score}`;
            isGameActive = true;
            container.innerHTML = '';
        }

        // Crear burbujas cada cierto tiempo
        setInterval(createBubble, 1000);
    </script>
